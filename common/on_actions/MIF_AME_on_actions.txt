on_actions = {
	#####################################
	### IMPEACHMENT CRISIS ON ACTIONS ###
	#####################################

	#Daily Variable Increases

	on_daily_AME = {
		if = {
			limit = {
				has_country_flag = AME_Impeachment_Crisis_flag
			}
			hidden_effect = {
				add_to_variable = {
					var = AME_Impeachment_Progress
					value = 0.01
				}
			}
		}
	}

	on_daily_AME = {
		if = {
			limit = {
				has_country_flag = AME_Impeachment_Crisis_flag
			}
			hidden_effect = {
				add_to_variable = {
					var = AME_Public_Outrage
					value = 0.01
				}
			}
		}
	}

	#Impeachment Progress and Public Outrage Variable Checks

	on_daily_AME = {
		if = {
			limit = {
				AND = {
					check_variable = {
						AME_Public_Outrage = 1
					}
					has_country_flag = AME_sided_with_congress_flag
				}
			}
			effect = {
				AME = {
					country_event = {
						id = impeachment.1 #McFarland Assassinated
					}
				}
			}
		}
	}

	on_daily_AME = {
		if = {
			limit = {
				AND = {
					check_variable = {
						AME_Public_Outrage = 1
					}
					has_country_flag = AME_sided_with_gair_flag
				}
			}
			effect = {
				AME = {
					country_event = {
						id = impeachment.2 #Gair Assassinated
					}
				}
			}
		}
	}

	on_daily_AME = {
		if = {
			limit = {
				AND = {
					check_variable = {
						AME_Impeachment_Progress = 1
					}
					has_country_flag = AME_sided_with_gair_flag
				}
			}
			effect = {
				AME = {
					country_event = {
						id = impeachment.3 #Gair Impeached - Gair Perspective
					}
				}
			}
		}
	}

	on_daily_AME = {
		if = {
			limit = {
				AND = {
					check_variable = {
						AME_Impeachment_Progress = 1
					}
					has_country_flag = AME_sided_with_congress_flag
				}
			}
			effect = {
				AME = {
					country_event = {
						id = impeachment.4 #Gair Impeached - McFarland Perspective
					}
				}
			}
		}
	}

	#Phase Clamps

	on_daily_AME = {
		effect = {
			if = {
				limit = {
					check_variable = {
						var = AME_Public_Outrage
						value = 0
						compare = greater_than_or_equals
					}
				}
				hidden_effect = {
					clamp_variable = {
						var = AME_Public_Outrage
						min = 0
						max = 1
					}
					set_country_flag = AME_Phase_One_Clamp
				}
			}
		}
	}

	on_daily_AME = {
		effect = {
			if = {
				limit = {
					AND = {
						check_variable = {
							var = AME_Public_Outrage
							value = 0.5
							compare = greater_than_or_equals
						}

						NOT = {
							check_variable = {
								var = AME_Public_Outrage
								value = 0.75
								compare = greater_than_or_equals
							}
						}
					}
				}
				hidden_effect = {
					clamp_variable = {
						var = AME_Public_Outrage
						min = 0.5
						max = 1
					}
					set_country_flag = AME_Phase_Two_Clamp
					if = {
						limit = {
							OR = {
								has_country_flag = AME_Phase_One_Clamp
							}
						}
						hidden_effect = {
							clr_country_flag = AME_Phase_One_Clamp
						}
					}
				}
			}
		}
	}

	on_daily_AME = {
		effect = {
			if = {
				limit = {
					check_variable = {
						var = AME_Public_Outrage
						value = 0.75
						compare = greater_than_or_equals
					}
				}
				hidden_effect = {
					clamp_variable = {
						var = AME_Public_Outrage
						min = 0.75
						max = 1
					}
					set_country_flag = AME_Phase_Three_Clamp
					if = {
						limit = {
							OR = {
								has_country_flag = AME_Phase_One_Clamp
								has_country_flag = AME_Phase_Two_Clamp
							}
						}
						hidden_effect = {
							clr_country_flag = AME_Phase_One_Clamp
							clr_country_flag = AME_Phase_Two_Clamp
						}
					}
				}
			}
		}
	}

	#Misc On Actions

	on_startup = {
		effect = {
			AME = {
				country_event = { 
					id = impeachment.5	#Impeachment Beginnings
					days = 379
				}
			}
		}
	}
}
